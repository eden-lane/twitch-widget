Twitch Widget

Displaying your stream status

Demo:


Using:

<head>
  <!-- Add jquery -->
  <script type="text/javascript" src="jquery.min.js"></script>
  <!-- Add styles -->
  <link rel="stylesheet" href="tw-widget.css"/>
</head>
<body>
  <!-- Add a div with 'tw-widget' class and name of your channel -->
  <div class="tw-widget" data-channel-name="YOUR_CHANNEL_NAME_HERE"></div>
  <!-- Add script -->
  <script type="text/javascript" src="tw-widget.min.js"></script>
</body>
